// program to blink the LED for 1 second and blow off for 1 second
// NOTE: here time is in milliseconds
// 1000 ms = 1 s

int LED = 13;
void setup()
{
  pinMode(LED, OUTPUT);
}

void loop()
{
  digitalWrite(LED, HIGH);
  delay(1000); // Wait for 1000 millisecond(s)
  digitalWrite(LED, LOW);
  delay(1000); // Wait for 1000 millisecond(s)
}
